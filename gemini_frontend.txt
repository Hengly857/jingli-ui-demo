 * * 核心技术展示：
 * 1. 【安全隔离】大模型 API 密钥锁死在后端，不暴露给任何客户端请求。
 * 2. 【指数级退避】集成弹性重试机制（Exponential Backoff），最大程度保证多模态调用稳定性。

/**
 * 弹性辅助函数：带有指数级退避重试（Exponential Backoff）的 fetch 请求封装
 * 严格遵循生产级故障复原要求，最大重试 5 次
 */
async function fetchWithRetry(url, options, retries = 5, delay = 1000) {
  try {
    const response = await fetch(url, options);
    if (!response.ok) {
      const errorText = await response.text();
      throw new Error(`HTTP ${response.status}: ${errorText}`);
    }
    return await response.json();
  } catch (error) {
    if (retries <= 1) throw error;
    // 指数级退避延迟计算: 1s, 2s, 4s, 8s, 16s...
    await new Promise(resolve => setTimeout(resolve, delay));
    return fetchWithRetry(url, options, retries - 1, delay * 2);
  }
}

    function convertPCMToWavBlob(base64PCM, sampleRate = 24000, numChannels = 1, bitsPerSample = 16) {
      // 1. 将 Base64 解码为二进制 ArrayBuffer
      const binaryString = window.atob(base64PCM);
      const len = binaryString.length;
      const bytes = new Uint8Array(len);
      for (let i = 0; i < len; i++) {
        bytes[i] = binaryString.charCodeAt(i);
      }
      
      const buffer = bytes.buffer;
      const wavHeaderBuffer = new ArrayBuffer(44);
      const view = new DataView(wavHeaderBuffer);
      
      /* RIFF 标识符 */
      writeString(view, 0, 'RIFF');
      /* 文件总大小 = 音频数据长度 + (44字节头部 - 8字节前缀占位) */
      view.setUint32(4, 36 + buffer.byteLength, true);
      /* WAVE 标识符 */
      writeString(view, 8, 'WAVE');
      /* Format Chunk 标识符 */
      writeString(view, 12, 'fmt ');
      /* Subchunk1Size = 16 (线性PCM格式) */
      view.setUint32(16, 16, true);
      /* AudioFormat = 1 (Linear PCM) */
      view.setUint16(20, 1, true);
      /* 声道数 (Mono = 1) */
      view.setUint16(22, numChannels, true);
      /* 采样率 (例如 24000Hz) */
      view.setUint32(24, sampleRate, true);
      /* ByteRate = SampleRate * NumChannels * BitsPerSample/8 */
      view.setUint32(28, sampleRate * numChannels * (bitsPerSample / 8), true);
      /* BlockAlign = NumChannels * BitsPerSample/8 */
      view.setUint16(32, numChannels * (bitsPerSample / 8), true);
      /* BitsPerSample = 16位 */
      view.setUint16(34, bitsPerSample, true);
      /* Data Chunk 标识符 */
      writeString(view, 36, 'data');
      /* Subchunk2Size (数据总长度) */
      view.setUint32(40, buffer.byteLength, true);
      
      // 合并 44字节 WAV 头文件和原始音频原始数据
      const combinedBlob = new Blob([wavHeaderBuffer, buffer], { type: 'audio/wav' });
